namespace MB09.Haustierverwaltung_1.Loesung
{
    public class Dog
    {
        public Dog(string name, int age)
        {
            Name = name;
            Age = age;
            Counter++;
        }

        public static int Counter { get; set; }

        public string Name { get; set; }

        public int Age { get; set; }
    }
}
